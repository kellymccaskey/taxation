\frame{\titlepage} 

\frame{\frametitle{Two things}
\begin{itemize}
\item Financial policy coordination increases the strength of migrant networks \pause 
\item The level of strength is different for different types of investment  
\end{itemize} 
}

\frame{\frametitle{Migrant networks \& international investment}
\begin{itemize}
\item Consequences of migration \pause 
\item Networks as information channels \pause
\item Migrant networks increase international investment \pause
\begin{itemize}
\item Portfolio investment \pause
\item Foreign direct investment 
\end{itemize}
\end{itemize} 
}

\frame{\frametitle{Why?}
\begin{itemize}
\item An increase in familiarity \pause 
\item A decrease in information asymmetries and transaction costs. 
\end{itemize} 
}

\frame{\frametitle{Structural barriers}
\begin{itemize}
\item Coordinated financial policies 
\begin{itemize}
\item Dual taxation treaties
\end{itemize}
\end{itemize} 
}

\frame{\frametitle{Hypotheses}
\begin{itemize}
\item Migrant networks have a larger effect on portfolio investment in combination with a dual taxation treaty than without\pause 
\item The combination of a dual taxation treaty and a migrant network has a negligible effect on FDI
\end{itemize} 
}
